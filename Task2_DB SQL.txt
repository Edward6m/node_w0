create table classes(
  class_id SERIAL PRIMARY KEY,
  class_name VARCHAR(20)
);
INSERT INTO
  classes(class_name)
VALUES
  ('三年一班'),
  ('三年二班');
create table student(
    student_id SERIAL PRIMARY KEY,
    student_name VARCHAR(20),
    gender VARCHAR(2),
    age INTEGER,
    class_id INTEGER,
    FOREIGN KEY (class_id) REFERENCES classes(class_id)
  );
INSERT INTO
  student(student_name, class_id, gender, age)
VALUES
  ('小明', 1, '男', 8),
  ('小華', 2, '女', 9),
  ('小美', 1, '男', 8),
  ('小強', 1, '女', 8),
  ('小智', 2, '男', 9);
select  *  FROM  classes;
select  *  FROM  student;
select 	s.student_id AS "學生編號",
  s.student_name AS "姓名", 
  s.gender AS "性別",
  s.age AS "年齡", c.class_name as 班級
from  student s
INNER JOIN classes c on s.class_id=c.class_id

CREATE TABLE parents(
 id SERIAL PRIMARY KEY,
 name VARCHAR(15),
 phone VARCHAR(10),
 gender VARCHAR(3)
);

CREATE TABLE children(
id SERIAL PRIMARY KEY,
name VARCHAR(15),
parent_id INTEGER,
FOREIGN KEY (parent_id) REFERENCES parents(id)
);

FROM children c
